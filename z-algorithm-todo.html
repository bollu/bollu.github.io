void mkz(const char *s, int *z) {
    const int N = strlen(s);
    int curi = 0;
    while(s[curi]) {
        int matchlen = 0; 
        while(curi + matchlen < N) {
            if (s[curi + matchlen] == s[matchlen]) { matchlen++; }
